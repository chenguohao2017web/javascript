const cookie = { set: function (key, val, time) { // 设置cookie方法 var date = new Date() // 获取当前时间 var expiresDays = time // 将date设置为n天以后的时间 date.setTime(date.getTime() + expiresDays * 24 * 3600 * 1000) // 格式化为cookie识别的时间 document.cookie = key + '=' + val + ';expires=' + date.toGMTString() // 设置cookie }, get: function (key) { // 获取cookie方法 /* 获取cookie参数 */ var search = key + '='// 查询检索的值 var returnvalue = ''// 返回值 if (document.cookie.length > 0) { var sd = document.cookie.indexOf(search) if (sd !== -1) { sd += search.length var end = document.cookie.indexOf(';', sd) if (end === -1) { end = document.cookie.length } // unescape() 函数可对通过 escape() 编码的字符串进行解码。 returnvalue = unescape(document.cookie.substring(sd, end)) } } return returnvalue } } export default cookie
